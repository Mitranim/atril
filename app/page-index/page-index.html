Custom Elements and Attributes 

// Viewmodel.

import {Component} from 'atril';

@Component({
  tagName: 'hello-world'
})
class ViewModel {
  name = 'world';
  static viewUrl = 'app/hello-world/hello-world.html';
}





Hello, {{name}}!















var Component = require('atril').Component;

Component({
  tagName: 'hello-world'
})(ViewModel);

function ViewModel() {
  this.name = 'world';
}

ViewModel.viewUrl = 'app/hello-world/hello-world.html';



Hello, {{name}}!
